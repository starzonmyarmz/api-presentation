Query Params

const params = new URLSearchParams(window.location.search)

has()

params.has('product') // true

get()

params.get('product') // 'tee shirt'

append()

params.append('qty', '5')
params.toString() // product=tee+shirt&color=blue&size=large&qty=5

set()

params.set('qty', '2')
params.toString() // product=tee+shirt&color=blue&size=large&qty=2

delete()

params.delete('qty')
params.toString() // product=tee+shirt&color=blue&size=large
